-- Enable UUID extension if not already enabled
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ==========================================
-- 1. CUSTOM TYPES & SEQUENCES
-- ==========================================
DO $$ BEGIN
    CREATE TYPE event_type_enum AS ENUM ('TIME', 'SCORE', 'DISTANCE', 'POINTS');
EXCEPTION
    WHEN duplicate_object THEN null;
END $$;

DO $$ BEGIN
    CREATE TYPE event_status_enum AS ENUM ('DRAFT', 'PUBLISHED');
EXCEPTION
    WHEN duplicate_object THEN null;
END $$;

CREATE SEQUENCE IF NOT EXISTS school_id_seq START 10000;
CREATE SEQUENCE IF NOT EXISTS student_id_seq START 10000;

-- ==========================================
-- 2. TABLES
-- ==========================================

-- A. Super Admin Table
CREATE TABLE IF NOT EXISTS public.super_admin (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    auth_id UUID REFERENCES auth.users(id) ON DELETE CASCADE UNIQUE,
    email TEXT UNIQUE NOT NULL,
    name TEXT NOT NULL DEFAULT 'Super Admin',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- B. Schools Table
CREATE TABLE IF NOT EXISTS public.schools (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    auth_id UUID REFERENCES auth.users(id) ON DELETE CASCADE UNIQUE,
    custom_school_id TEXT UNIQUE,
    school_name TEXT NOT NULL,
    principal_name TEXT,
    email TEXT UNIQUE NOT NULL,
    phone TEXT,
    district TEXT,
    state TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- C. Students Table
CREATE TABLE IF NOT EXISTS public.students (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    auth_id UUID REFERENCES auth.users(id) ON DELETE CASCADE UNIQUE,
    school_id UUID REFERENCES public.schools(id) ON DELETE SET NULL,
    custom_student_id TEXT UNIQUE,
    full_name TEXT NOT NULL,
    gender TEXT,
    dob DATE,
    email TEXT UNIQUE NOT NULL,
    phone TEXT,
    photo_url TEXT,
    district TEXT,
    state TEXT,
    points INTEGER DEFAULT 0, -- GAMIFICATION POINTS COLUMN
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- D. Events Table
CREATE TABLE IF NOT EXISTS public.events (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT NOT NULL,
    sport_category TEXT,
    logo_url TEXT,
    description TEXT,
    event_date DATE NOT NULL,
    venue TEXT NOT NULL,
    age_category TEXT NOT NULL,
    reg_start_date DATE NOT NULL,
    reg_end_date DATE NOT NULL,
    max_attempts INT DEFAULT 1,
    rules TEXT,
    event_type event_type_enum NOT NULL, -- Defines ranking algorithm
    status event_status_enum DEFAULT 'DRAFT',
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- E. Event Participants (Registration)
CREATE TABLE IF NOT EXISTS public.event_participants (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    event_id UUID REFERENCES public.events(id) ON DELETE CASCADE,
    student_id UUID REFERENCES public.students(id) ON DELETE CASCADE,
    reg_date DATE DEFAULT CURRENT_DATE,
    reg_time TIME DEFAULT CURRENT_TIME,
    status TEXT DEFAULT 'REGISTERED',
    UNIQUE(event_id, student_id) -- Prevent double registration
);

-- F. Performances
CREATE TABLE IF NOT EXISTS public.performances (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    event_id UUID REFERENCES public.events(id) ON DELETE CASCADE,
    student_id UUID REFERENCES public.students(id) ON DELETE CASCADE,
    metric_value NUMERIC NOT NULL, -- Represents seconds, points, etc.
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(event_id, student_id) -- One active performance per student per event
);

-- G. Rankings (Auto-calculated via Trigger)
CREATE TABLE IF NOT EXISTS public.rankings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    event_id UUID REFERENCES public.events(id) ON DELETE CASCADE,
    student_id UUID REFERENCES public.students(id) ON DELETE CASCADE,
    performance_id UUID REFERENCES public.performances(id) ON DELETE CASCADE,
    metric_value NUMERIC NOT NULL,
    air_rank INT NOT NULL,
    state_rank INT NOT NULL,
    district_rank INT NOT NULL,
    last_calculated TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(event_id, student_id)
);


-- ==========================================
-- 3. TRIGGERS FOR AUTO-GENERATED IDs
-- ==========================================

-- Generate SCH ID
CREATE OR REPLACE FUNCTION generate_school_id() RETURNS TRIGGER AS $$
BEGIN
    NEW.custom_school_id := 'SCH' || LPAD(nextval('school_id_seq')::text, 6, '0');
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_generate_school_id ON public.schools;
CREATE TRIGGER trg_generate_school_id
BEFORE INSERT ON public.schools
FOR EACH ROW EXECUTE FUNCTION generate_school_id();

-- Generate AKI Student ID (Format: AKI + Year + Sequence)
CREATE OR REPLACE FUNCTION generate_student_id() RETURNS TRIGGER AS $$
BEGIN
    NEW.custom_student_id := 'AKI' || TO_CHAR(CURRENT_DATE, 'YYYY') || LPAD(nextval('student_id_seq')::text, 5, '0');
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_generate_student_id ON public.students;
CREATE TRIGGER trg_generate_student_id
BEFORE INSERT ON public.students
FOR EACH ROW EXECUTE FUNCTION generate_student_id();


-- ==========================================
-- 4. AUTOMATED RANKING ENGINE (THE CORE LOGIC)
-- ==========================================
CREATE OR REPLACE FUNCTION recalculate_event_rankings() RETURNS TRIGGER AS $$
DECLARE
    v_event_type event_type_enum;
BEGIN
    SELECT event_type INTO v_event_type FROM public.events WHERE id = NEW.event_id;

    DELETE FROM public.rankings WHERE event_id = NEW.event_id;

    INSERT INTO public.rankings (event_id, student_id, performance_id, metric_value, air_rank, state_rank, district_rank)
    SELECT 
        p.event_id,
        p.student_id,
        p.id AS performance_id,
        p.metric_value,
        RANK() OVER (
            ORDER BY 
                CASE WHEN v_event_type = 'TIME' THEN p.metric_value END ASC,
                CASE WHEN v_event_type != 'TIME' THEN p.metric_value END DESC
        ) AS air_rank,
        RANK() OVER (
            PARTITION BY s.state 
            ORDER BY 
                CASE WHEN v_event_type = 'TIME' THEN p.metric_value END ASC,
                CASE WHEN v_event_type != 'TIME' THEN p.metric_value END DESC
        ) AS state_rank,
        RANK() OVER (
            PARTITION BY s.state, s.district 
            ORDER BY 
                CASE WHEN v_event_type = 'TIME' THEN p.metric_value END ASC,
                CASE WHEN v_event_type != 'TIME' THEN p.metric_value END DESC
        ) AS district_rank
    FROM public.performances p
    JOIN public.students s ON p.student_id = s.id
    WHERE p.event_id = NEW.event_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_recalculate_rankings ON public.performances;
CREATE TRIGGER trg_recalculate_rankings
AFTER INSERT OR UPDATE ON public.performances
FOR EACH ROW EXECUTE FUNCTION recalculate_event_rankings();

CREATE POLICY "Anyone can view rankings" ON public.rankings FOR SELECT USING (true);

-- ==========================================
-- 6. SUPER ADMIN BYPASS
-- ==========================================
CREATE OR REPLACE FUNCTION is_admin() RETURNS BOOLEAN AS $$
BEGIN
  RETURN EXISTS (SELECT 1 FROM public.super_admin WHERE auth_id = auth.uid());
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;
